\setcounter{i}{1}
\makeatletter
\whiledo{\value{i} < 100}{ % Busca hasta el archivo 99.tex

        \input{content/\the\value{i}}

        \stepcounter{i}
    }{
        \setcounter{i}{101} % Se detiene si falta un número en la secuencia
    }
}
\makeatother
